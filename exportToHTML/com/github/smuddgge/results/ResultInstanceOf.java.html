3    /**
4     * Used in a {@link ResultChecker} to check if the
5     * value is an instance of another class
6     */
7    public class ResultInstanceOf implements Result {
8    
9        /**
10        * The class used to check the instance of the value
11        */
12       private final Class object;
13   
14       /**
15        * Used to create a new {@link ResultInstanceOf}
16        *
17        * @param object The class
18        * @param <T>    The type of class
19        */
20       public <T> ResultInstanceOf(Class<T> object) {
21           this.object = object;
22       }
23   
24       @Override
25       public boolean check(Object value) {
26           return object.isAssignableFrom(value.getClass());
27       }
28   }
