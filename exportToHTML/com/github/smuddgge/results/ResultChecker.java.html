5    /**
6     * <h2>Represents a result checker</h2>
7     * Used to check results to make sure they are correct
8     */
9    public class ResultChecker {
10   
11       /**
12        * Used to check a {@link Result} type
13        *
14        * @param value  Value to check
15        * @param result Result to check against
16        * @return This instance to chain multiple checks
17        */
18       public ResultChecker expect(Object value, Result result) throws Exception {
19           if (!result.check(value))
20               throw new Exception(value + " != " + result.getClass().getName());
21           return this;
22       }
23   
24       /**
25        * Used to check an exact value
26        *
27        * @param value Value to check
28        * @param match Value to match
29        * @return This instance to chain multiple checks
30        */
31       public ResultChecker expect(Object value, Object match) throws Exception {
32           if (!Objects.equals(value, match))
33               throw new Exception(value + " != " + match);
34           return this;
35       }
36   }
