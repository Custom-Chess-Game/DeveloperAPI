3    /**
4     * <h2>A fields value type</h2>
5     * Used to specify what type of value the field is
6     * so databases such as sqlite get identify what type the field is.
7     */
8    public enum FieldValueType {
9        STRING, INTEGER;
10   
11       /**
12        * Used to get a field value type from a class instance
13        *
14        * @param instance Instance of a class
15        * @return Null if it is not a field value type
16        */
17       public static FieldValueType get(Class<?> instance) {
18           if (instance.isAssignableFrom(String.class)) return FieldValueType.STRING;
19           if (instance.isAssignableFrom(Integer.class)) return FieldValueType.INTEGER;
20           return null;
21       }
22   }
