1    package com.github.smuddgge.console;
2    
3    /**
4     * <h2>Represents the console</h2>
5     * Using the {@link Console} class to log instead of
6     * system outs will help organise debugging statements
7     * from logging statements.
8     */
9    public class Console {
10   
11       /**
12        * <h2>Log a message in the console</h2>
13        * You can use the {@link ConsoleColour} enum in the message
14        * string to add colour.
15        *
16        * @param message Message to log
17        */
18       public static void log(String message) {
19           System.out.println(ConsoleColour.parse(ConsoleColour.RESET + "[LOG] " + message));
20       }
21   
22       /**
23        * <h2>State a warning in the console</h2>
24        * This type of logging is used when errors occur that are
25        * not fatal to the program.
26        *
27        * @param message Message to log
28        */
29       public static void warn(String message) {
30           System.out.println(ConsoleColour.parse(ConsoleColour.RESET + "[WARN] " + message));
31       }
32   
33       /**
34        * <h2>State an error in console</h2>
35        * Can be used in a try catch method when the error is fatal
36        * to the program.
37        *
38        * @param message Message to log
39        */
40       public static void error(String message) {
41           System.out.println(ConsoleColour.parse(ConsoleColour.RESET + "" + ConsoleColour.YELLOW + "[ERROR] " + message));
42       }
43   }
