3    /**
4     * <h2>Represents a colour displayed in the console</h2>
5     * This enum {@link ConsoleColour} can be used in {@link Console}
6     * print methods to add colour in the console.
7     */
8    public enum ConsoleColour {
9        RESET, GRAY, GREEN, YELLOW, PINK;
10   
11       /**
12        * Used to parse a message colours
13        *
14        * @param message Message to parse
15        * @return Parsed message
16        */
17       public static String parse(String message) {
18           return message
19                   .replace("RESET", "\033[0m")
20                   .replace("GRAY", "\033[0;37m")
21                   .replace("GREEN", "\033[0;32m")
22                   .replace("YELLOW", "\033[0;33m")
23                   .replace("PINK", "\033[0;35m");
24       }
25   }
